例子

class X {
    private final Lock lock = new ReentrantLock();
    public void lock() {
        lock.lock();
        try {
            // do something
        } finally {
            lock.unlock();
        }
    }
}

非公平锁

final void lock() {
    if (compareAndSetState(0, 1))
        //成功则设置独占线程为当前线程
        setExclusiveOwnerThread(Thread.currentThread());
    else
        //失败则去调用aqs的acquire方法
        acquire(1);
}
  • AbstractQueuedSynchronizer的acquire方法
    public final void acquire(int arg) {
  //子类实现tryAcquire方法，tryAcquire调用了NonfairSync的父类Sync的nonfairTryAcquire方法（下面的 *1）
  if (!tryAcquire(arg) &&
      // 如果tryAcquire失败，则不间断的去获取排队的线程（下面的 *2和 *3）
      acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
      selfInterrupt();
}
  • *1 Sync的nonfairTryAcquire方法
    /**
* 先判断state=0的话，直接cas设置state，成功则设置独占线程为当前线程
* state!=0的话，如果独占线程为当前线程，则state+1，返回true
* 否则返回false
*/
final boolean nonfairTryAcquire(int acquires) {
  final Thread current = Thread.currentThread();
  int c = getState();
  if (c == 0) {
      if (compareAndSetState(0, acquires)) {
          setExclusiveOwnerThread(current);
          return true;
      }
  }
  else if (current == getExclusiveOwnerThread()) {
      int nextc = c + acquires;
      if (nextc < 0) // overflow
          throw new Error("Maximum lock count exceeded");
      setState(nextc);
      return true;
  }
  return false;
}
  • *2 AbstractQueuedSynchronizer的addWaiter方法
    private Node addWaiter(Node mode) {
  Node node = new Node(Thread.currentThread(), mode);
  //如果队尾tail不为空，则把该node置为队尾
  Node pred = tail;
  if (pred != null) {
      node.prev = pred;
      if (compareAndSetTail(pred, node)) {
          pred.next = node;
          return node;
      }
  }
  // 队尾tail为null时，把该node入队
  enq(node);
  return node;
}

private Node enq(final Node node) {
// 自旋方式保证加入到队尾
for (;;) {
Node t = tail;
if (t == null) {
//tail为null时，初始化head和tail
if (compareAndSetHead(new Node()))
tail = head;
} else {
//cas设置tail
node.prev = t;
if (compareAndSetTail(t, node)) {
t.next = node;
return t;
}
}
}
}

- *3 AbstractQueuedSynchronizer的acquireQueued方法

final boolean acquireQueued(final Node node, int arg) {
boolean failed = true;
try {
boolean interrupted = false;
for (;;) {
//获取该node的前继节点
final Node p = node.predecessor();
//前继节点是head且能以独占方式获取成功，则把node设置为head，返回
if (p == head && tryAcquire(arg)) {
setHead(node);
p.next = null; // help GC
failed = false;
return interrupted;
}
//失败则检查是否需要挂起（下面的*4）
if (shouldParkAfterFailedAcquire(p, node) &&
//需要挂起的话，则执行挂起动作
parkAndCheckInterrupt())
interrupted = true;
}
} finally {
if (failed)
//如果acquire失败了，则去掉acquire
cancelAcquire(node);
}
}

- *4 AbstractQueuedSynchronizer的shouldParkAfterFailedAcquire方法

private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
int ws = pred.waitStatus;
//SIGNAL表示该node的后继者的线程需要唤醒
if (ws == Node.SIGNAL)
//pred是SIGNAL状态时，说明node是需要阻塞挂起的
return true;
if (ws > 0) {
//状态>0表示CANCELLED，是说线程已被取消
//因此需要一直往前找节点，知道节点的waitStatus不大于0
do {
node.prev = pred = pred.prev;
} while (pred.waitStatus > 0);
pred.next = node;
} else {
//如果前继节点不是CANCELLED，也不是SIGNAL状态，则把它的状态设置为SIGNAL
compareAndSetWaitStatus(pred, ws, Node.SIGNAL);
}
return false;
}

private final boolean parkAndCheckInterrupt() {
//挂起当前线程
LockSupport.park(this);
//返回是否被打断
return Thread.interrupted();
}


### 公平锁
- 与非公平锁的差别在于tryAcquire方法的不同

protected final boolean tryAcquire(int acquires) {
final Thread current = Thread.currentThread();
int c = getState();
if (c == 0) {
//如果state=0的话，首先要判断是不是有等待获取锁的节点
//如果没有的话，则try成功
//如果有的话，返回try失败
if (!hasQueuedPredecessors() &&
compareAndSetState(0, acquires)) {
setExclusiveOwnerThread(current);
return true;
}
}
//如果独占线程就是当前线程，则可以直接try成功
else if (current == getExclusiveOwnerThread()) {
int nextc = c + acquires;
if (nextc < 0)
throw new Error(“Maximum lock count exceeded”);
setState(nextc);
return true;
}
return false;
}

// 如果有排队的线程早于当前线程，则返回true
// 如果当前线程在队列的head处，或者队列是空的，则返回false
public final boolean hasQueuedPredecessors() {
Node t = tail;
Node h = head;
Node s;
return h != t &&
((s = h.next) == null || s.thread != Thread.currentThread());
}

// AQS的release方法
public final boolean release(int arg) {
//子类Sync的tryRelease（下面的 *1 ）
if (tryRelease(arg)) {
Node h = head;
if (h != null && h.waitStatus != 0)
//tryRelease成功则唤醒下一个节点（下面的 *2 ）
unparkSuccessor(h);
return true;
}
return false;
}


- *1 Sync的tryRelease方法

protected final boolean tryRelease(int releases) {
int c = getState() - releases;
//只能持有锁的才能释放锁
if (Thread.currentThread() != getExclusiveOwnerThread())
throw new IllegalMonitorStateException();
boolean free = false;
//c=0是指所有的锁都释放了，自然要把独占线程置为null
if (c == 0) {
free = true;
setExclusiveOwnerThread(null);
}
setState(c);
return free;
}


- *2 AQS的unparkSuccessor方法

private void unparkSuccessor(Node node) {
int ws = node.waitStatus;
if (ws < 0)
compareAndSetWaitStatus(node, ws, 0);
//通常是head的下一个节点，如果head的下一个节点是null或者状态是CANCELLED，
//那么就要从tail一直往前找不是CANCELLED的节点
Node s = node.next;
if (s == null || s.waitStatus > 0) {
s = null;
for (Node t = tail; t != null && t != node; t = t.prev)
if (t.waitStatus <= 0)
s = t;
}
//唤醒这个节点
if (s != null)
LockSupport.unpark(s.thread);
}
